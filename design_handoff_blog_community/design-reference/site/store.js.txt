// tone rotation so tag chips & covers feel varied but on-brand
const TAG_TONES = ['flame', 'teal', 'honey', 'ink'];
function toneForTag(tag) {
  let h = 0;
  for (let i = 0; i < tag.length; i++) h = (h * 31 + tag.charCodeAt(i)) >>> 0;
  return TAG_TONES[h % TAG_TONES.length];
}

// avatar fallback colors (deterministic from a string)
const AVATAR_TONES = [
  { bg: 'var(--orange-50)', fg: 'var(--orange-700)' },
  { bg: 'var(--teal-50)', fg: 'var(--teal-700)' },
  { bg: 'var(--honey-100)', fg: 'var(--honey-700)' },
  { bg: 'var(--neutral-100)', fg: 'var(--neutral-700)' },
];
function toneForName(name) {
  let h = 0;
  for (let i = 0; i < (name || '').length; i++) h = (h * 31 + name.charCodeAt(i)) >>> 0;
  return AVATAR_TONES[h % AVATAR_TONES.length];
}
function initials(name) {
  return (name || '?').trim().split(/\s+/).slice(0, 2).map((s) => s[0] || '').join('').toUpperCase();
}

const uid = () => Math.random().toString(36).slice(2, 10);
function readMinutes(html) {
  const words = (html || '').replace(/<[^>]+>/g, ' ').trim().split(/\s+/).length;
  return Math.max(1, Math.round(words / 200));
}

function loadState() {
  try {
    const raw = localStorage.getItem(LS_KEY);
    if (raw) {
      const parsed = JSON.parse(raw);
      if (parsed && parsed.members && parsed.posts) {
        // migrate older saves that predate later collections
        if (!parsed.signups) parsed.signups = seedSignups();
        if (!parsed.events) parsed.events = seedEvents();
        if (!parsed.rsvps) parsed.rsvps = seedRsvps(parsed.events);
        if (!parsed.volunteers) parsed.volunteers = seedVolunteers();
        if (!parsed.emails) parsed.emails = seedEmails(parsed.events, parsed.members, parsed.signups);
        if (!parsed.comments) parsed.comments = seedComments();
        if (!parsed.reactions) parsed.reactions = seedReactions();
        if (!parsed.myReactions) parsed.myReactions = {};
        if (!parsed.feedback) parsed.feedback = seedFeedback();
        if (!parsed.analytics) parsed.analytics = seedAnalytics();
        return parsed;
      }
    }
  } catch (e) { /* ignore */ }
  // first run — seed, and sign the owner in as admin so the review experience is full
  const events = seedEvents();
  const members = seedMembers();
  const signups = seedSignups();
  return { members, posts: seedPosts(), signups, events, rsvps: seedRsvps(events),
    volunteers: seedVolunteers(), emails: seedEmails(events, members, signups), comments: seedComments(),
    reactions: seedReactions(), myReactions: {}, feedback: seedFeedback(), analytics: seedAnalytics(), authUserId: 'u_admin' };
}

  const parts = String(ev.date || '').split(' ');
  return new Date(parseInt(ev.year || '2026', 10), MONTH_IDX[parts[0]] || 0, parseInt(parts[1] || '1', 10));

const Store = {
  state: loadState(),
  listeners: new Set(),
  persist() {
    try { localStorage.setItem(LS_KEY, JSON.stringify(this.state)); } catch (e) { /* ignore */ }
  },
  set(updater) {
    const next = typeof updater === 'function' ? updater(this.state) : updater;
    this.state = { ...this.state, ...next };
    this.persist();
    this.listeners.forEach((l) => l());
  },
  subscribe(fn) { this.listeners.add(fn); return () => this.listeners.delete(fn); },

  // ---- selectors ----
  currentUser() { return this.state.members.find((m) => m.id === this.state.authUserId) || null; },
  memberById(id) { return this.state.members.find((m) => m.id === id) || null; },
  postById(id) { return this.state.posts.find((p) => p.id === id) || null; },
  publishedPosts() {
    return this.state.posts.filter((p) => p.status === 'published')
      .sort((a, b) => (a.date < b.date ? 1 : -1));
  },
  pendingPosts() {
    return this.state.posts.filter((p) => p.status === 'pending')
      .sort((a, b) => (a.date < b.date ? 1 : -1));
  },
  allTags() {
    const set = new Set(TAG_SUGGESTIONS);
    this.state.posts.forEach((p) => (p.tags || []).forEach((t) => set.add(t)));
    return Array.from(set);
  },

  // ---- signups / events / rsvps selectors ----
  signups() { return (this.state.signups || []).slice().sort((a, b) => (a.date < b.date ? 1 : -1)); },
  eventById(id) { return (this.state.events || []).find((e) => e.id === id) || null; },
  sortedEvents() { return (this.state.events || []).slice(); },
  rsvpsForEvent(id) { return (this.state.rsvps || []).filter((r) => r.eventId === id).sort((a, b) => (a.date < b.date ? 1 : -1)); },
  allRsvps() { return (this.state.rsvps || []).slice().sort((a, b) => (a.date < b.date ? 1 : -1)); },

  // ---- auth ----
  signUp({ name, email }) {
    const existing = this.state.members.find((m) => m.email.toLowerCase() === (email || '').toLowerCase());
    if (existing) { this.set({ authUserId: existing.id }); return existing; }
    const member = {
      id: uid(), name: name || '', email: email || '', photo: null,
      role: '', company: '', linkedin: '', website: '', location: '', admin: false,
      bio: '', expertise: [], joined: new Date().toISOString().slice(0, 10), fresh: true,
    };
    this.set((s) => ({ members: [...s.members, member], authUserId: member.id }));
    return member;
  },
  logIn(email) {
    const m = this.state.members.find((x) => x.email.toLowerCase() === (email || '').toLowerCase());
    if (m) { this.set({ authUserId: m.id }); return m; }
    return null;
  },
  logOut() { this.set({ authUserId: null }); },

  // ---- members ----
  updateMember(id, patch) {
    this.set((s) => ({ members: s.members.map((m) => (m.id === id ? { ...m, ...patch, fresh: false } : m)) }));
  },
  setMemberRole(id, role) { this.updateMember(id, { role }); },
  setAdmin(id, admin) { this.updateMember(id, { admin }); },
  removeMember(id) {
    this.set((s) => ({ members: s.members.filter((m) => m.id !== id) }));
  },

  // ---- posts ----
  submitPost({ title, excerpt, body, tags, cover, authorId }) {
    const post = {
      id: uid(), authorId, title, excerpt, body, tags: tags || [], cover: cover || null,
      date: new Date().toISOString().slice(0, 10), status: 'pending', readMins: readMinutes(body),
    };
    this.set((s) => ({ posts: [post, ...s.posts] }));
    return post;
  },
  setPostStatus(id, status) {
    this.set((s) => ({ posts: s.posts.map((p) => (p.id === id ? { ...p, status } : p)) }));
  },
  deletePost(id) { this.set((s) => ({ posts: s.posts.filter((p) => p.id !== id) })); },

  // ---- mailing-list signups ----
  addSignup({ name, email, source }) {
    const entry = { id: uid(), name: name || '', email: email || '', source: source || 'Mailing list', date: new Date().toISOString().slice(0, 10) };
    this.set((s) => ({ signups: [entry, ...(s.signups || [])] }));
    return entry;
  },

  // ---- events (admin-managed) ----
  createEvent(data) {
    const ev = { id: uid(), status: 'open', year: String(new Date().getFullYear()), ...data };
    this.set((s) => ({ events: [ev, ...(s.events || [])] }));
    this.announceEvent(ev);
    return ev;
  },
  updateEvent(id, patch) { this.set((s) => ({ events: (s.events || []).map((e) => (e.id === id ? { ...e, ...patch } : e)) })); },
  deleteEvent(id) { this.set((s) => ({ events: (s.events || []).filter((e) => e.id !== id), rsvps: (s.rsvps || []).filter((r) => r.eventId !== id) })); },

  // ---- rsvps ----
  addRSVP({ eventId, name, email, guests }) {
    const r = { id: uid(), eventId, name: name || '', email: email || '', guests: guests || 1, date: new Date().toISOString().slice(0, 10) };
    this.set((s) => ({ rsvps: [r, ...(s.rsvps || [])] }));
    this.scheduleReminder(r, this.eventById(eventId));
    return r;
  },

  // ---- volunteers ----
  addVolunteer({ name, email, company, position, role }) {
    const v = { id: uid(), name: name || '', email: email || '', company: company || '', position: position || '', role: role || '', date: new Date().toISOString().slice(0, 10) };
    this.set((s) => ({ volunteers: [v, ...(s.volunteers || [])] }));
    return v;
  },
  volunteers() { return (this.state.volunteers || []).slice().sort((a, b) => (a.date < b.date ? 1 : -1)); },

  // ---- email (outbox / log) ----
  recipientsAll() {
    const seen = new Set(); const out = [];
    (this.state.members || []).forEach((m) => { const e = (m.email || '').toLowerCase(); if (e && !seen.has(e)) { seen.add(e); out.push({ name: m.name, email: m.email }); } });
    (this.state.signups || []).forEach((s) => { const e = (s.email || '').toLowerCase(); if (e && !seen.has(e)) { seen.add(e); out.push({ name: s.name, email: s.email }); } });
    return out;
  },
  emails() { return (this.state.emails || []).slice(); },
  announceEvent(ev) {
    const recipients = this.recipientsAll();
    const email = { id: uid(), type: 'announcement', subject: 'New event: ' + ev.title, eventId: ev.id,
      body: `You're invited! ${ev.title} happens ${ev.day} ${ev.date}, ${ev.year} at ${ev.time}, ${ev.where}.\n\n${ev.desc || ''}\n\nAdd it to your calendar and we'll see you there.`,
      recipients, count: recipients.length, status: 'sent', date: new Date().toISOString().slice(0, 10) };
    this.set((s) => ({ emails: [email, ...(s.emails || [])] }));
    return email;
  },
  scheduleReminder(rsvp, ev) {
    if (!ev) return null;
    const email = { id: uid(), type: 'reminder', subject: 'Reminder: ' + ev.title + ' is tomorrow', eventId: ev.id,
      body: `A friendly reminder that ${ev.title} is happening ${ev.day} ${ev.date} at ${ev.time}, ${ev.where}. We saved your spot — see you there!`,
      recipients: [{ name: rsvp.name, email: rsvp.email }], count: 1, status: 'scheduled', scheduledFor: reminderDate(ev), date: new Date().toISOString().slice(0, 10) };
    this.set((s) => ({ emails: [email, ...(s.emails || [])] }));
    return email;
  },
  sendBlast({ subject, body, audience }) {
    let recipients;
    if (audience === 'members') recipients = (this.state.members || []).map((m) => ({ name: m.name, email: m.email }));
    else if (audience === 'list') recipients = (this.state.signups || []).map((s) => ({ name: s.name, email: s.email }));
    else recipients = this.recipientsAll();
    const email = { id: uid(), type: 'blast', subject, body, audience: audience || 'all', recipients, count: recipients.length, status: 'sent', date: new Date().toISOString().slice(0, 10) };
    this.set((s) => ({ emails: [email, ...(s.emails || [])] }));
    return email;
  },
  markEmailSent(id) { this.set((s) => ({ emails: (s.emails || []).map((e) => (e.id === id ? { ...e, status: 'sent' } : e)) })); },

  // ---- reactions (visitor-friendly) ----
  reactionsFor(postId) { return (this.state.reactions || {})[postId] || {}; },
  myReacted(postId, type) { return !!(this.state.myReactions || {})[postId + ':' + type]; },
  toggleReaction(postId, type) {
    this.set((s) => {
      const reactions = { ...(s.reactions || {}) }; const counts = { ...(reactions[postId] || {}) };
      const mine = { ...(s.myReactions || {}) }; const key = postId + ':' + type;
      if (mine[key]) { delete mine[key]; counts[type] = Math.max(0, (counts[type] || 0) - 1); }
      else { mine[key] = true; counts[type] = (counts[type] || 0) + 1; }
      reactions[postId] = counts;
      return { reactions, myReactions: mine };
    });
  },

  // ---- comments (visitor-friendly) ----
  commentsForPost(postId) { return (this.state.comments || []).filter((c) => c.postId === postId).sort((a, b) => (a.date < b.date ? 1 : -1)); },
  addComment({ postId, authorId, name, text }) {
    const c = { id: uid(), postId, authorId: authorId || null, name: name || 'Anonymous', text: text || '', date: new Date().toISOString() };
    this.set((s) => ({ comments: [c, ...(s.comments || [])] }));
    return c;
  },
  deleteComment(id) { this.set((s) => ({ comments: (s.comments || []).filter((c) => c.id !== id) })); },

  // ---- feedback (visitor-friendly) ----
  addFeedback({ rating, message, email, page }) {
    const f = { id: uid(), rating: rating || 0, message: message || '', email: email || '', page: page || '', date: new Date().toISOString() };
    this.set((s) => ({ feedback: [f, ...(s.feedback || [])] }));
    return f;
  },
  feedbackList() { return (this.state.feedback || []).slice().sort((a, b) => (a.date < b.date ? 1 : -1)); },

  // ---- analytics / engagement ----
  analytics() { return this.state.analytics || {}; },
  trackPostView(postId) {
    this.set((s) => {
      const a = { ...(s.analytics || {}) };
      a.postViews = { ...(a.postViews || {}) };
      a.postViews[postId] = (a.postViews[postId] || 0) + 1;
      a.pageviews = (a.pageviews || 0) + 1;
      return { analytics: a };
    });
  },
  postEngagement(postId) {
    const views = (this.analytics().postViews || {})[postId] || 0;
    const reacts = Object.values(this.reactionsFor(postId)).reduce((n, v) => n + v, 0);
    const comments = this.commentsForPost(postId).length;
    return { views, reacts, comments, score: views + reacts * 8 + comments * 15 };
  },

  resetAll() {
    const events = seedEvents();
    const members = seedMembers();
    const signups = seedSignups();
    this.state = { members, posts: seedPosts(), signups, events, rsvps: seedRsvps(events),
      volunteers: seedVolunteers(), emails: seedEmails(events, members, signups), comments: seedComments(),
      reactions: seedReactions(), myReactions: {}, feedback: seedFeedback(), analytics: seedAnalytics(), authUserId: 'u_admin' };
    this.persist();
    this.listeners.forEach((l) => l());
  },
};

// React hook — re-renders subscribers on any store change
function useStore() {
  const [, force] = React.useReducer((n) => n + 1, 0);
  React.useEffect(() => Store.subscribe(force), []);
  return Store;
}
